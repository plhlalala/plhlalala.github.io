Sigaction 挑战性任务实验报告

题目概览

sigaction 是 Unix 与类 Unix 操作系统中进程间或者内核与进程间的一种 异步通信，用来提醒一个进程某一信号已经发生。

当一个信号被发送给一个进程时, 内核会中断进程的正常控制流，转而 执行与该信号相关的用户态处理函数 进行处理。在执行该处理函数前，会将该信号所设置的信号屏蔽集加入到进程的信号屏蔽集中。在执行完该用户态处理函数后，又会将恢复原来的信号屏蔽集。

实现思路

sigaction结构体

typedef struct sigset_t {
    uint32_t sig;
} sigset_t;

struct sigaction {
    void     (*sa_handler)(int);
    sigset_t   sa_mask;
}

sigaction 用于存储一个信号对应的处理函数以及执行时需要添加到当前进程的掩码。

env结构体

struct Env {
    // challenge
    struct sigaction env_sigs[33];
    sigset_t env_sa_mask;
    uint32_t env_sig_list;
    u_int env_sig_entry;
};
  • env_sigs 用于存储当前进程对应的 32 个信号的处理函数以及掩码。
  • env_sa_mask 对应当前进程的掩码。
  • env_sig_list 有 32 位，分别对应当前进程有哪些信号在等待处理，其中一种信号在未被处理的时候来了多个只算一个。
  • env_sig_entry 是对应的用户态处理统一处理的函数入口。

用户态下的信号掩码处理函数

以下是信号掩码处理函数的实现：

int sigemptyset(sigset_t *set) {
    if (set == NULL) return -1;
    set->sig = 0;
    return 0;
}

int sigfillset(sigset_t *set) {
    if (set == NULL) return -1;
    set->sig = 0xffffffff;
    return 0;
}

int sigaddset(sigset_t *set, int _signo) {
    if (set == NULL || _signo <= 0 || _signo > 32) return -1;
    set->sig |= (1 << (_signo - 1));
    return 0;
}

int sigdelset(sigset_t *set, int _signo) {
    if (set == NULL || _signo <= 0 || _signo > 32) return -1;
    set->sig &= (~(1 << (_signo - 1)));
    return 0;
}

int sigismember(const sigset_t *set, int _signo) {
    if (set == NULL || _signo <= 0 || _signo > 32) return -1;
    return (set->sig & (1 << (_signo - 1))) ? 1 : 0;
}

int sigisemptyset(const sigset_t *set) {
    if (set == NULL) return -1;
    return (set->sig == 0) ? 1 : 0;
}

int sigandset(sigset_t *set, const sigset_t *left, const sigset_t *right) {
    if (set == NULL || left == NULL || right == NULL) return -1;
    set->sig = left->sig & right->sig;
    return 0;
}

int sigorset(sigset_t *set, const sigset_t *left, const sigset_t *right) {
    if (set == NULL || left == NULL || right == NULL) return -1;
    set->sig = left->sig | right->sig;
    return 0;
}
  • sigemptyset 用于初始化信号集，将其置为空。
  • sigfillset 用于将所有信号加入信号集。
  • sigaddset 用于向信号集中添加特定信号。
  • sigdelset 用于从信号集中删除特定信号。
  • sigismember 用于检查特定信号是否在信号集中。
  • sigisemptyset 用于检查信号集是否为空。
  • sigandset 用于获取两个信号集的交集。
  • sigorset 用于获取两个信号集的并集。

需要注意的是要能够正确处理非法参数，比如 NULL。

用户态下的信号注册以及进程相关函数

以下是信号注册和相关操作的函数实现：

int sigaction(int signum, const struct sigaction *newact, struct sigaction *oldact) {
    if (signum > 32 || signum <= 0) {
        return -1;
    }
    if (signum == 9) {
        return -1;
    }
    return syscall_sigaction(signum, newact, oldact);
}

int kill(u_int envid, int sig) {
    if (sig <= 0 || sig > 32) {
        return -1;
    }
    return syscall_kill(envid, sig);
}

int sigprocmask(int __how, const sigset_t *__set, sigset_t *__oset) {
    return syscall_proc_mask(__how, __set, __oset, 0);
}

int sigpending(sigset_t *__set) {
    if (__set == NULL) return -1;
    (__set->sig) = env->env_sa_mask.sig & env->env_sig_list;
    (__set->sig) &= (~(1 << 8));
    return 0;
}
  • sigaction 用于注册信号处理函数。
  • kill 用于向某个进程发送信号。
  • sigprocmask 用于修改或获取当前进程的信号掩码。
  • sigpending 用于获取当前进程被阻塞且未处理的信号。

相关系统调用

以下是相关系统调用的实现：

int sys_set_sig_entry(u_int envid, u_int func) {
    struct Env *env;
    try(envid2env(envid, &env, 0));
    env->env_sig_entry = func;
    return 0;
}

int sys_sigaction(int signum, const struct sigaction *newact, struct sigaction *oldact) {
    if (oldact) {
        *oldact = curenv->env_sigs[signum];
    }
    if (newact) {
        curenv->env_sigs[signum] = *newact;
        curenv->env_sigs[signum].sa_mask.sig &= (~(1 << 8));
    }
    return 0;
}

int sys_killl(u_int envid, int sig) {
    struct Env *env;
    if (sig < 1 || sig > 32) {
        return -1;
    }
    try(envid2env(envid, &env, 0));
    env->env_sig_list |= 1 << (sig - 1);
    return 0;
}

int sys_proc_mask(int how, sigset_t *set, sigset_t *oldset, int signum) {
    if (oldset != NULL) {
        oldset->sig = curenv->env_sa_mask.sig;
    }
    if (how != SIG_BLOCK && how != SIG_SETMASK && how != SIG_UNBLOCK) {
        return -1;
    }
    if (set != NULL) {
        switch (how) {
            case SIG_BLOCK:
                curenv->env_sa_mask.sig |= set->sig;
                curenv->env_sa_mask.sig &= (~(1 << 8));
                break;
            case SIG_UNBLOCK:
                curenv->env_sa_mask.sig &= (~(set->sig));
                break;
            case SIG_SETMASK:
                curenv->env_sa_mask.sig = set->sig;
                curenv->env_sa_mask.sig &= (~(1 << 8));
                break;
            default:
                break;
        }
    }
    if (signum != 0) {
        curenv->env_sa_mask.sig |= (1 << (signum - 1));
    }
    return 0;
}

用户态统一处理函数入口

void __attribute__((noreturn)) sig_entry(struct Trapframe *tf, void (*sa_handler)(int), int signum, sigset_t *set) {
    sigset_t tmpsig;
    syscall_proc_mask(SIG_BLOCK, set, &tmpsig, signum);
    switch (signum) {
        case SIGINT:
        case SIGILL:
        case SIGSEGV:
            if (!sa_handler) {
                exit();
            }
            break;
        case SIGKILL:
            exit();
            break;
        case SIGSYS:
            if (!sa_handler) {
                tf->cp0_epc += 4;
            }
            break;
        default:
            break;
    }
    if (sa_handler) {
        sa_handler(signum);
    }
    syscall_proc_mask(SIG_SETMASK, &tmpsig, NULL, 0);
    int r = syscall_set_trapframe(0, tf);
    user_panic("sighand_entry returned %d", r);
}

这个函数 sig_entry 是信号处理函数的入口，当进程收到信号后，内核会跳转到这个函数进行处理。

主要功能是：

  1. 阻塞当前信号集。

  2. 根据信号类型做相应的处理。

  3. 调用用户定义的信号处理函数。

  4. 恢复原来的信号掩码。

  5. 恢复进程的上下文，并继续执行。

void __attribute__((noreturn)) sig_entry(struct Trapframe *tf, void (*sa_handler)(int), int signum, sigset_t *set) {
  • __attribute__((noreturn)) 表示这个函数不会返回。
  • 函数参数：
    • tf：保存当前的 CPU 寄存器状态。
    • sa_handler：信号处理函数。
    • signum：信号编号。
    • set：信号屏蔽集。
sigset_t tmpsig;
syscall_proc_mask(SIG_BLOCK, set, &tmpsig, signum);
  • 定义一个临时的信号集 tmpsig。
  • 调用 syscall_proc_mask 系统调用阻塞 set 中的信号，并保存当前的信号掩码到 tmpsig。
switch (signum) {
    case SIGINT:
    case SIGILL:
    case SIGSEGV:
        if (!sa_handler) {
            exit();
        }
        break;
    case SIGKILL:
        exit();
        break;
    case SIGSYS:
        if (!sa_handler) {
            tf->cp0_epc += 4;
        }
        break;
    default:
        break;
}
  • 根据信号编号 signum 进行处理：
    • 对于 SIGINT、SIGILL 和 SIGSEGV，如果没有信号处理函数 sa_handler，则调用 exit() 终止进程。
    • 对于 SIGKILL，直接调用 exit() 终止进程。
    • 对于 SIGSYS，如果没有信号处理函数 sa_handler，则将程序计数器 cp0_epc 向前移动4字节，跳过非法的系统调用指令。
if (sa_handler) {
    sa_handler(signum);
}
  • 如果存在信号处理函数 sa_handler，则调用该处理函数，并传入信号编号 signum。
syscall_proc_mask(SIG_SETMASK, &tmpsig, NULL, 0);
  • 调用 syscall_proc_mask 恢复原来的信号掩码，即将 tmpsig 设置为当前的信号掩码。
int r = syscall_set_trapframe(0, tf);
user_panic("sighand_entry returned %d", r);
  • 调用 syscall_set_trapframe 系统调用，设置进程的 Trapframe 为 tf，恢复进程的上下文。
  • 如果 syscall_set_trapframe 返回值不为0，则调用 user_panic 输出错误信息，并终止进程。

sig_entry 函数作为信号处理的入口，首先阻塞当前信号集，然后根据信号类型做相应处理，如果有用户定义的信号处理函数则调用之，处理完毕后恢复原来的信号掩码，最后恢复进程的上下文并继续执行。

信号处理函数的跳转

FEXPORT(ret_from_exception)
    move    a0, sp
    addiu   sp, sp, -8
    jal     do_signal
    nop
    addiu   sp, sp, 8
    RESTORE_ALL
    eret
void do_signal(struct Trapframe *tf) {
    if (((int)tf->cp0_epc) > ULIM) {
        return;
    }
    uint32_t available = curenv->env_sig_list & (~(curenv->env_sa_mask).sig);
    int signum = 0;
    for (int i = 1; i <= 32; i++) {
        if (available & (1 << (i - 1))) {
            signum = i;
            break;
        }
    }
    if (curenv->env_sig_list & (1 << 8)) {
        signum = 9;
    }
    if (signum == 0) {
        return;
    }
    if (curenv->env_sig_entry) {
        struct Trapframe tmp_tf = *tf;
        if (tf->regs[29] < USTACKTOP || tf->regs[29] >= UXSTACKTOP) {
            tf->regs[29] = UXSTACKTOP;
        }
        tf->regs[29] -= sizeof(struct Trapframe);
        *((struct Trapframe *)tf->regs[29]) = tmp_tf;
        curenv->env_sig_list &= ~(1 << (signum - 1));
        tf->regs[4] = tf->regs[29];
        tf->regs[5] = (unsigned int)(curenv->env_sigs[signum].sa_handler);
        tf->regs[6] = signum;
        tf->regs[7] = &(curenv->env_sigs[signum].sa_mask);
        tf->regs[29] -= sizeof(tf->regs[4]);
        tf->regs[29] -= sizeof(tf->regs[5]);
        tf->regs[29] -= sizeof(tf->regs[6]);
        tf->regs[29] -= sizeof(tf->regs[7]);
        tf->cp0_epc = curenv->env_sig_entry;
    }
}

这个函数 do_signal 负责处理信号。当一个进程接收到信号时，内核会调用这个函数来处理该信号。

函数目的

该函数的主要目的是在收到信号时，保存当前的CPU状态，将信号处理函数的参数传递给用户态的信号处理函数，并跳转到信号处理函数进行执行。

代码详解

void do_signal(struct Trapframe *tf) {
    if (((int)tf->cp0_epc) > ULIM) {
        return;
    }
  1. 检查异常程序计数器 (EPC):
    • tf->cp0_epc 保存了发生异常时的程序计数器的值。
    • ULIM 是用户空间和内核空间的边界。如果 tf->cp0_epc 超过了 ULIM，说明发生异常时正在执行内核代码，不需要处理信号，直接返回。
uint32_t available = curenv->env_sig_list & (~(curenv->env_sa_mask).sig);
  1. 计算当前可用处理的信号:
    • curenv->env_sig_list 表示当前等待处理的信号。
    • curenv->env_sa_mask 表示当前屏蔽的信号。
    • available 保存了当前等待处理且未被屏蔽的信号。
int signum = 0;
for (int i = 1; i <= 32; i++) {
    if (available & (1 << (i - 1))) {
        signum = i;
        break;
    }
}
  1. 查找第一个需要处理的信号:
    • 遍历信号编号1到32，找到第一个未被屏蔽且需要处理的信号，将其编号赋值给 signum。
if (curenv->env_sig_list & (1 << 8)) {
    signum = 9;
}
  1. 处理特殊信号:
    • 如果当前有信号9（SIGKILL）在信号列表中，无论屏蔽与否，立即处理该信号。
if (signum == 0) {
    return;
}
  1. 检查是否有需要处理的信号:
    • 如果 signum 为0，表示没有需要处理的信号，直接返回。
    if (curenv->env_sig_entry) {
        struct Trapframe tmp_tf = *tf;
        if (tf->regs[29] < USTACKTOP || tf->regs[29] >= UXSTACKTOP) {
            tf->regs[29] = UXSTACKTOP;
        }
        tf->regs[29] -= sizeof(struct Trapframe);
        *((struct Trapframe *)tf->regs[29]) = tmp_tf;
        curenv->env_sig_list &= ~(1 << (signum - 1));
        tf->regs[4] = tf->regs[29];
        tf->regs[5] = (unsigned int)(curenv->env_sigs[signum].sa_handler);
        tf->regs[6] = signum;
        tf->regs[7] = &(curenv->env_sigs[signum].sa_mask);
        tf->regs[29] -= sizeof(tf->regs[4]);
        tf->regs[29] -= sizeof(tf->regs[5]);
        tf->regs[29] -= sizeof(tf->regs[6]);
        tf->regs[29] -= sizeof(tf->regs[7]);
        tf->cp0_epc = curenv->env_sig_entry;
    }
}
  1. 处理信号:
    • 如果 curenv->env_sig_entry 存在，表示有用户态的信号处理函数。
    • 保存当前 Trapframe 到用户栈中。
      • tf->regs[29] 是栈指针寄存器（SP），如果它位于用户栈范围内，则设置为用户异常栈顶 UXSTACKTOP。
      • 将 Trapframe 结构体保存到用户栈中。
    • 更新信号列表，清除当前处理的信号位。
    • 设置信号处理函数的参数：
      • tf->regs[4] 指向保存的 Trapframe。
      • tf->regs[5] 指向信号处理函数。
      • tf->regs[6] 为信号编号。
      • tf->regs[7] 为信号屏蔽集。
    • 调整栈指针，以便保存这些参数。
    • 设置程序计数器 tf->cp0_epc 为信号处理入口地址，准备跳转到信号处理函数。

do_signal 函数的作用是在收到信号时，保存当前的CPU状态，将信号处理函数的参数传递给用户态的信号处理函数，并跳转到信号处理函数进行执行。如果当前没有需要处理的信号，或者信号在内核态触发，则直接返回。

fork的处理

fork时子进程需要继承所有sigaction，当前进程掩码，以及sig_entry。

进程信号统一处理函数入口的设置

void libmain(int argc, char **argv) {
    env = &envs[ENVX(syscall_getenvid())];
    syscall_set_sig_entry(0, sig_entry);
    main(argc, argv);
    exit();
}

我对于设置sig_entry采取的是在libos.c的libmain中设置入口，在fork时继承入口，即可保证任何用户进程有正确的用户处理函数。

完成时遇到的一些问题

以下是一个测试程序：

#include <lib.h>

void sigchld_handler(int sig) {
    debugf("capture SIGCHLD signal.\n");
    debugf("shouldn't see this.\n");
}

int main() {
    struct sigaction sa;
    sa.sa_handler = sigchld_handler;
    sigemptyset(&sa.sa_mask);
    sigaction(SIGCHLD, &sa, NULL);
    u_int pid = syscall_getenvid();
    if (fork() == 0) {
        kill(pid, SIGKILL);
        exit();
    }
    while (1);
    return 0;
}

对于这个测试程序，可能会表现出来如下的输出:

[00001001] destroying 00001001
[00001001] free env 00001001
i am killed ... 
capture SIGCHLD signal.
shouldn't see this.
[00000800] destroying 00000800
[00000800] free env 00000800
i am killed ...

由于 do_signal 先处理 SIGKILL，但是在跳转到 sig_entry 的时候发生了缺页异常，这个时候位于用户空间的缺页异常 ret_from_exception 的时候会重入 do_signal，就会去处理 SIGCHLD，表现出来的就是先进入 SIGCHLD 的处理程序，然后再执行 SIGKILL 的默认 exit 处理，这个好像是可以接受的，符合对于处理信号时机的要求。
